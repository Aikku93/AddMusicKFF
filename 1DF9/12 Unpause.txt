#jsr changePause2 $DA $02 $0C $54 $B4 $B0 $B4 $18 $B0

#asm changePause2
{
	mov	a, !SpeedUpBackUp	;\
	mov	$0387, a			;/ Restore the tempo.

	mov a, #$00
	mov	!PauseMusic, a
	
	mov a, $d1		;\
	mov y, #$2c		;|
	call DSPWrite	;| Restore echo volumes from backup
	mov a, $d2		;|
	mov y, #$3c		;|
	call DSPWrite	;/
	
	mov	a, #$7f		;\
	mov	y, #$0c		;| Set left master volume to $7f
	call DSPWrite	;/
	mov	a, #$7f		;\
	mov	y, #$1c		;| Set right master volume to $7f
	jmp	DSPWrite	;/
}
