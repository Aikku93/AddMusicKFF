#jsr changePause

#asm changePause
{
	mov	a, !SpeedUpBackUp	;\
	mov	$0387, a			;/ Restore the tempo.
	
	mov	$f2, #$5c		; \ Key off voices
	mov	$f3, #$ff		; / (so the music doesn't restart playing when using start+select)
	
	mov	a, #$7f		;\
	mov	y, #$0c		;| Set left master volume to $7f
	call DSPWrite	;/
	mov	a, #$7f		;\
	mov	y, #$1c		;| Set right master volume to $7f
	jmp	DSPWrite	;/
}
